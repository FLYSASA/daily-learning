### 1. 以下分别打印什么？ 

见vue轮子省市区插播面试题：
```js
// 1.
for(var i = 0; i < 2; i++){
  console.log(i)     // 0 1
}

// 2.
// 用var定义的i, 全局只要一个，for循环最后 ++， 所以全局变量i为2，所以最后打印都为2
for(var i = 0; i < 2; i++){
  setTimeout(()=>{
    console.log(i)  // 2 2
  })
}

// 3.
// 用 let ,每次进入循环重新声明了一个
for(let i = 0; i < 2; i++){
  // let i = _i // 保证每次进来都是赋值进来的
  setTimeout(()=>{
    console.log(i)  // 0 1
  })
}

// 4.
// let a = 0,1  // a = 1
// let a = (0,1)  // a = 1
for (
  // 这里只是提供了一个初始值0，与后面 i < 2; i++ 不再有关联
  let i = (setTimeout(()=> {console.log(i)}, 2000),0);  
  i < 2;
  i++
) {
  console.log(i)      // 0 1, 2s后打印 0
}


// 5.
for (
  var i = (setTimeout(()=> {console.log(i)}, 2000),0);
  i < 2;
  i++
) {
  console.log(i)      // 0 1, 2s后打印 2
}
```
